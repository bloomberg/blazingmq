423 {
424  // PRECONDITIONS
425  BSLS_ASSERT_OPT(8 * 1024 * 1024 < value);
426  // We reserve 4MB for control message (see
427  // 'bmqimp::BrokerSession::k_CONTROL_DATA_WATERMARK_EXTRA') so make
428  // sure the provided value is greater than it.
429

431  return *this;
